#!/usr/bin/env python3 """ Actualización manual de datos.json — Plan B si el scraping falla. Modo de uso: python3 actualizar_manual.py Te pide los porcentajes uno por uno y reescribe datos.json. Útil para el día E si el scraper de Registraduría no responde. También puedes usarlo en modo "batch" pasando un archivo plano: python3 actualizar_manual.py --archivo input.txt donde input.txt tiene este formato: cep 39.5 del 32.1 pal 18.2 """ import argparse import json import sys from datetime import datetime from pathlib import Path ROOT_DIR = Path(__file__).resolve().parent.parent DATOS_PATH = ROOT_DIR / "datos.json" CANDIDATOS_NOMBRES = { "cep": "Iván Cepeda", "del": "Abelardo de la Espriella", "pal": "Paloma Valencia", "faj": "Sergio Fajardo", "lop": "Claudia López", } def cargar_datos(): with open(DATOS_PATH, "r", encoding="utf-8") as f: return json.load(f) def guardar_datos(datos): with open(DATOS_PATH, "w", encoding="utf-8") as f: json.dump(datos, f, ensure_ascii=False, indent=2) print(f"✓ Guardado en {DATOS_PATH}") def modo_interactivo(datos): print("\n" + "=" * 60) print("ACTUALIZACIÓN MANUAL DE PORCENTAJES") print("=" * 60) print("Para cada candidato escribe el nuevo porcentaje y presiona Enter.") print("Deja vacío para mantener el valor actual.\n") for c in datos["primera_vuelta"]["candidatos"]: actual = c["porcentaje"] nuevo = input(f" {c['nombre']:30s} actual={actual:5.1f}% → nuevo: ").strip() if nuevo: try: nuevo_val = float(nuevo.replace(",", ".")) c["cambio_dia"] = round(nuevo_val - actual, 1) c["porcentaje"] = nuevo_val print(f" → cambio: {c['cambio_dia']:+.1f} pts") except ValueError: print(f" ✗ '{nuevo}' no es un número válido. Se mantiene {actual}%") fuente = input(f"\nFuente (default 'Registraduría · Manual'): ").strip() datos["meta"]["fuente_principal"] = fuente or "Registraduría · Manual" datos["meta"]["fecha_actualizacion"] = datetime.now().strftime("%Y-%m-%d") datos["meta"]["hora_actualizacion"] = datetime.now().strftime("%H:%M") datos["meta"]["fase"] = "resultados_oficiales" confirmar = input("\n¿Guardar cambios? (s/N): ").strip().lower() if confirmar == "s": guardar_datos(datos) else: print("✗ Cancelado, no se guardó nada.") def modo_archivo(datos, ruta_archivo): print(f"Leyendo {ruta_archivo}...") with open(ruta_archivo, "r", encoding="utf-8") as f: for linea in f: linea = linea.strip() if not linea or linea.startswith("#"): continue partes = linea.split() if len(partes) != 2: print(f" ⚠ Línea ignorada: {linea}") continue cid, valor = partes try: nuevo_val = float(valor.replace(",", ".")) except ValueError: print(f" ⚠ Valor inválido: {valor}") continue for c in datos["primera_vuelta"]["candidatos"]: if c["id"] == cid: c["cambio_dia"] = round(nuevo_val - c["porcentaje"], 1) c["porcentaje"] = nuevo_val print(f" ✓ {cid}: {nuevo_val}% (cambio {c['cambio_dia']:+.1f})") break else: print(f" ⚠ ID no encontrado: {cid}") datos["meta"]["fuente_principal"] = "Registraduría · Manual (batch)" datos["meta"]["fecha_actualizacion"] = datetime.now().strftime("%Y-%m-%d") datos["meta"]["hora_actualizacion"] = datetime.now().strftime("%H:%M") guardar_datos(datos) def main(): parser = argparse.ArgumentParser() parser.add_argument("--archivo", help="Ruta a archivo plano con porcentajes") args = parser.parse_args() datos = cargar_datos() if args.archivo: modo_archivo(datos, args.archivo) else: modo_interactivo(datos) if __name__ == "__main__": main()
